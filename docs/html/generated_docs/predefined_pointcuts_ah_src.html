 1#ifndef __ACP__PREDEFINED_POINTCUTS_AH__

 4/**
 5 * \collection PredefinedPointcuts
 6 * \file src/core/predefined_pointcuts.ah
 7 * \brief This file contains predefined pointcuts that can be used in aspects.
 8 *
 9 * These pointcuts are designed to match various types of functions in a C++ project.
10 * Its purpose is to provide a set of commonly used pointcuts that can be reused across
11 * different aspects. Further increasing the user friendliness of this library in the case
12 * a user wants to create their own aspects.
13 */
14namespace acp {
15	/**
16	 * \pointcut all_func
17	 * \brief Matches all functions in the whole project.
18	 */
19	pointcut all_func() = "% ...::%(...)";
20
21	/**
22	 * \pointcut free_func
23	 * \brief Matches all free functions in the whole project.
24	 *
25	 * Free functions are functions that are not part of a class or namespace.
26	 */
27    pointcut free_func() = "% %(...)";
28
29	/**
30	 * \pointcut static_func
31	 * \brief Matches all static functions in the whole project.
32	 */
33    pointcut static_func() = "static % ...::%(...)";
34
35	/**
36	 * \pointcut returns_primitive
37	 * \brief Matches all functions that return a primitive data type.
38	 */
39    pointcut returns_primitive() = "int ...::%(...)" || "long ...::%(...)" || "short ...::%(...)" || "char ...::%(...)" || "float ...::%(...)" || "double ...::%(...)" || "bool ...::%(...)";
40
41	/**
42	 * \pointcut returns_void
43	 * \brief Matches all functions that return void.
44	 */
45    pointcut returns_void() = "void ...::%(...)";
46
47	/**
48	 * \pointcut returns_pointer
49	 * \brief Matches all functions that return a pointer.
50	 */
51    pointcut returns_pointer() = "%* ...::%(...)";
52
53	/**
54	 * \pointcut returns_reference
55	 * \brief Matches all functions that return a reference.
56	 */
57    pointcut returns_reference() = "%& ...::%(...)";
58
59	/**
60	 * \pointcut returns_const
61	 * \brief Matches all functions that return a const value.
62	 */
63    pointcut returns_const() = "const % ...::%(...)";
64
65    /**
66	 * \pointcut main_func
67	 * \brief Matches every possible argument variant of the c++ main function.
68	 */
69	pointcut main_function() = "int main(...)";
70
71	/**
72	 * \pointcut setters
73	 * \brief Matches all setter functions in the whole project.
74	 */
75    pointcut setters() = "void set%(...)";
76
77    /**
78	 * \pointcut getters
79	 * \brief Matches all getter functions in the whole project.
80	 */
81    pointcut getters() = "% get%(...)";
82
83	/**
84	 * \pointcut none
85	 * \brief Matches no join points.
86	 */
87	pointcut none() = "void crifmpamoolnisklrqmx()";
88
89}
