 14namespace acp {
 15	/**
 16	 * \attribute acp::trace
 17	 * \brief Triggers the appearance of a join point in the case of an uncaught exception in the stacktrace.
 18	 *
 19	 * Useless by default, since the aspect is applied to every join point by default.
 20	 */
 21	attribute trace();
 22
 23	/**
 24	 * \attribute acp::fallback
 25	 * \brief Triggers the printing of the stacktrace if a thrown exception leaves the scope of a join point.
 26	 *
 27	 * Specially recommended to use this attribute in multithreaded applications, because the standard fall-
 28	 * back function "main()"" is never reached for all but one thread.
 29	 */
 30	attribute fallback();
 31
 32	/**
 33	 * \attribute acp::ignore_trace
 34	 * \brief This attribute can be used to exclude certain join points form being affected by this aspect.
 35     *
 36     * If a scope already is annotated with this aspect's attribute, individual join points within can be
 37     * excluded with this attribute.
 38	 *
 39	 * Not recommend to use this, because of the same reason that the trace attribute is currently unused:
 40	 * By excluding join points from the stacktrace the resulting stacktrace will be incomplete.
 41	 */
 42	attribute ignore_trace();
 43}

 45/**
 46 * \aspect StackTrace
 47 * \file src/error_handling/stack_trace.ah
 48 * \brief This aspect is used to handle stack traces and print them when an uncaught exception occurs.
 49 *
 50 * This aspect collects the recent call history in the event of a thrown exception. When the exception is
 51 * caught eventually or changes to a different one, the corrected data will be discarded.
 52 * Upon an thrown exception reaching and leaving a join point marked with the acp::fallback attribute the
 53 * stack trace will be printed.
 54 * The message itself is shaped by the "build_msg()" function. You can replace its content easily by
 55 * writing an aspect to replace it.
 56 * The preferred printing device, which is provided to the constructor, is used to make an output.
 57 */
 58aspect [[acp::outer_layer]] StackTrace : public CoreAspect {
 59
 60private:
 61
 62	// Useing this wrapper function to create static threadlocal header only variable in c++11
 63	inline bool& editing_stack_trace_instance() {
 64		static thread_local bool _editing_stack_trace = false;
 65		return _editing_stack_trace;
 66	}
 67
 68	// Useing this wrapper function to create static threadlocal header only variable in c++11
 69	inline std::exception const*& last_thrown_instance() {
 70		static thread_local std::exception const* _last_thrown = nullptr;
 71		return _last_thrown;
 72	}
 73
 74	// Useing this wrapper function to create static threadlocal header only variable in c++11
 75	inline std::vector<std::string>& call_chain_instance() {
 76		static thread_local std::vector<std::string> _call_chain{};
 77		return _call_chain;
 78	}
 79
 80	acp::Printer* _printer;
 81
 82	void check_reset() {
 83		// When calling a function, the exception must have been resolved
 84		if (last_thrown_instance() != nullptr and !editing_stack_trace_instance()) {
 85			last_thrown_instance() = nullptr;
 86			call_chain_instance().clear();
 87		}
 88	}
 89
 90protected:
 91
 92	/**
 93	 * \function print_stack_trace
 94	 * \brief This method handles everything related to printing the stack trace.
 95	 *
 96	 * In case a developer wants to change the way the stack trace is printed from the ground up,
 97	 * they can write an around advice for this function to replace its countent.
 98	 *
 99	 * \param call_chain is a std::vector of strings that contains the call chain of the stack trace.
100	 * \param exception_msg is the message of the exception that was thrown.
101	 */
102	void print_stack_trace(const std::vector<std::string>& call_chain, const char* exception_msg) {
103		_printer->print("\n\nUncaught exception ");
104		_printer->print(exception_msg);
105		_printer->print(":\n");
106		for (const auto& err_msg : call_chain_instance()) {
107			_printer->print('\t');
108			_printer->print(err_msg);
109		}
110		// Reset values after printing.
111		last_thrown_instance() = nullptr;
112		_printer->print("\n");
113		editing_stack_trace_instance() = false;
114	}
115
116	/**
117	 * \function build_msg
118	 * \brief This methods is responsible for creating a well formatted stack trace message
119	 *
120	 * In case you want to change this, write an aspect which replaces this function with your
121	 * own implementation. Use a around advice for that.
122	 *
123	 * \param sign is the signature of the function that was called.
124	 * \param file is the name of the file where the function was called.
125	 * \param line is the line number where the function was called.
126	 * \param result is a reference to a string where the formatted message will be stored.
127	 */
128	inline static void build_msg(const char* sign, const char* file, int line, std::string& result) {
129        char buffer[12];
130        acp::to_string(buffer, line, sizeof(buffer));
131        result.clear();
132        result.append("at ");
133        result.append(sign);
134        result.append(" in (");
135        result.append(file);
136        result.append(":");
137        result.append(buffer);
138        result.append(")\n");
139	}

170	/**
171	 * \advice fallback-advice
172	 * \brief This advice handles the StackTrace printing when an uncaught exception leaves the function of the pointcut.
173	 */
174	advice fallback_func() && !ignore() : around() {
175
176		check_reset();
177		try {
178			tjp->proceed();
179		} catch (const std::exception& e) {
180			editing_stack_trace_instance() = true;
181			std::string msg;
182			build_msg(tjp->signature(), tjp->filename(), tjp->line(), msg);
183			call_chain_instance().push_back(msg);
184			// Print Stack Trace
185			print_stack_trace(call_chain_instance(), e.what());
186			editing_stack_trace_instance() = false;
187			// Exit the program with code 0. Should be something different, but tests otherwise fail.
188			std::exit(0);
189		} catch (...) {
190			// Non std::exceptions not implemented.
191		}
192	}
193
194	/**
195	 * \advice stack-trace-builder-advice
196	 * \brief Advice handles the StackTrace building, in case of an thrown exception.
197	 */
198	advice where() && !fallback_func() && !ignore() : around(){
199
200		check_reset();
201		try {
202			tjp->proceed();
203		} catch (const std::exception& e) {
204			editing_stack_trace_instance() = true;
205			// extra variable initialization, to supress warning in else-if condition
206			const auto& last_thrown = last_thrown_instance();
207			if (last_thrown == nullptr) {
208				// Just thrown
209				std::string msg;
210				build_msg(tjp->signature(),tjp->filename(),tjp->line(),msg);
211				call_chain_instance().push_back(msg);
212				last_thrown_instance() = &e;
213			} else if (last_thrown != nullptr and typeid(*last_thrown) == typeid(e) and std::string(last_thrown_instance()->what()) == std::string(e.what())) {
214				// Exception is still the same
215				std::string msg;
216				build_msg(tjp->signature(),tjp->filename(),tjp->line(),msg);
217				call_chain_instance().push_back(msg);
218			} else {
219				// Exception changed
220				call_chain_instance().clear();
221				std::string msg;
222				build_msg(tjp->signature(),tjp->filename(),tjp->line(),msg);
223				call_chain_instance().push_back(msg);
224				last_thrown_instance() = &e;
225			}
226			editing_stack_trace_instance() = false;
227			throw;
228		} catch (...) {
229			// Non std::exceptions not implemented.
230		}
231	}
