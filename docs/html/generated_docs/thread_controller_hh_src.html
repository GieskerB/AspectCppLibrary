 8    /**
 9     * \interface ThreadController
10     * \file src/core/thread_controller.hh
11     * \brief Interface for thread control mechanisms.
12     *
13     * This interface provides methods for thread safety, including acquiring and releasing locks,
14     * and getting the unique identifier for the current thread.
15     *
16     * To statically configure the thread controller at runtime, use an aspect that implements this interface.
17     * Example being the `acp::std_thread_controller`.
18     *
19     */
20    namespace thread_controller {
21        /**
22         * \function acquire_lock
23         * \brief Acquire a lock for thread safety.
24         *
25         * This method should be called before accessing shared resources to ensure
26         * that no other thread can modify them concurrently.
27         */
28        inline void acquire_global_lock() {}
29
30        /**
31         * \function release_lock
32         * \brief Release the lock acquired by acquire_lock.
33         *
34         * This method should be called after finishing access to shared resources
35         * to allow other threads to acquire the lock.
36         */
37        inline void release_global_lock(){}
38
39        /**
40         * \function synchronized_function_call
41         * \brief Calls the argument given by the first argument with mutex exclusion.
42         *
43         * \param func and std::function which should be called.
44         */
45        inline void synchronized_function_call(const std::function<void()>& func) {
46            func();
47        }
48
49        /**
50         * \function get_id
51         * \brief Get the unique identifier for the current thread.
52         *
53         * This method returns a unique identifier for the current thread, which can
54         * be used for logging or debugging purposes.
55         *
56         * \return An integer representing the thread ID.
57         */
58        inline int get_id(){return 0;}
59
60    };
61}
