 37/**
 38 * \aspect Logging
 39 * \file src/functionality/logging.ah
 40 * \brief This aspect is used to log the execution of a function.
 41 *
 42 * With the default configuration of this aspect the function signature will be printed alongside the
 43 * "action" relative to the function: Entering od leaving.
 44 *
 45 * This aspect supports three different attributes: acp::after_logging, acp::around_logging and
 46 * acp::before_logging. Depending on the choosen attribute the logging happens at the expected point
 47 * relative to the execution.
 48 * The messages themself are shaped by the "build_msg()", "build_before_msg()" and "build_after_msg"
 49 * function. You can replace its content easily by writing an aspect to replace it.
 50 * The preferred printing device, which is provided to the constructor, is used to make an output.
 51 */
 52aspect [[acp::inner_layer]] Logging : public CoreAspect {
 53
 54    acp::Printer* _printer;
 55
 56protected:
 57
 58    /**
 59     * \function build_msg
 60     * \brief This function is responsible for creating a well formatted general logging message.
 61     *
 62     * In case you want to change this, write an aspect which replaces this function with your
 63	 * own implementation. Use a around advice for that.
 64     *
 65     * \param log_type The type of log message (e.g., "INFO", "ERROR", etc.).
 66     * \param msg The message to log.
 67     * \param output The output string where the built message will be stored.
 68     */
 69    inline static void build_msg(const std::string& log_type, const std::string& msg, std::string& output) {
 70        output.clear();
 71        if (log_type.empty()) {
 72            output.append(msg);
 73        } else if (msg.empty()) {
 74            output.append("[");
 75            output.append(log_type);
 76            output.append("]");
 77        } else {
 78            output.append("[");
 79            output.append(log_type);
 80            output.append("] ");
 81            output.append(msg);
 82        }
 83    };
 84
 85    /**
 86     * \function build_before_msg
 87     * \brief This function is responsible for creating a well formatted entering logging message.
 88     *
 89     * In case you want to change this, write an aspect which replaces this function with your
 90	 * own implementation. Use a around advice for that.
 91     *
 92     * \param signature The signature of the function being logged.
 93     * \param output The output string where the built message will be stored.
 94     */
 95    inline static void build_before_msg(const char* signature, std::string& output) {
 96        output.clear();
 97        output.append("[Entering]");
 98        output.append(signature);
 99    }
100
101    /**
102     * \function build_after_msg
103     * \brief This function is responsible for creating a well formatted leaving logging message.
104     *
105     * In case you want to change this, write an aspect which replaces this function with your
106	 * own implementation. Use a around advice for that.
107     *
108     * \param signature The signature of the function being logged.
109     * \param output The output string where the built message will be stored.
110     */
111    inline static void build_after_msg(const char* signature, std::string& output) {
112        output.clear();
113        output.append("[Leaving]");
114        output.append(signature);
115    }
116
117public:
118
119     /**
120	 * \pointcut ignore
121	 * \see :ref:`CoreAspect::ignore <core_aspect_ah_ignore>`
122	 */
123    pointcut virtual ignore() = within("Logging") || within ("acp") || execution(acp::ignore_logging()) || execution(acp::ignore_everything());
124
125    /**
126	 * \pointcut where
127	 * \see :ref:`CoreAspect::where <core_aspect_ah_where>`
128     *
129     * This aspect comes with three different attributes all having a similar effect. This is not
130     * foreseen in the design idea of this library. Therefore, this aspect only has two always
131     * active attributes and only one "configurable": acp::around_logging.
132	 */
133    pointcut virtual where() = execution(acp::around_logging());
134
135    Logging(): Logging(nullptr) {}
136    Logging(acp::Printer* printer)
137        : _printer(printer) {}
138
139    virtual ~Logging() {
140        delete _printer;
141    }
142
143    /**
144     * \advice logging-advice[after]
145     * \brief This advice handles the logging after the function executes.
146     */
147    advice execution(acp::after_logging()) && !ignore() : after() {
148        std::string msg;
149        build_after_msg(tjp->signature(), msg);
150        _printer->print(msg);
151        _printer->print('\n');
152    }
153
154    /**
155     * \advice logging-advice[around]
156     * \brief This advice handles the logging before and after the function executes.
157     *
158     * Notice that this advice uses the where join point which is by default set to acp::around:logging.
159     */
160    advice where() && !ignore() : around() {
161        std::string msg;
162        build_before_msg(tjp->signature(), msg);
163        _printer->print(msg);
164        _printer->print('\n');
165        tjp->proceed();
166        build_after_msg(tjp->signature(), msg);
167        _printer->print(msg);
168        _printer->print('\n');
169    }
170
171    /**
172     * \advice logging-advice[before]
173     * \brief This advice handles the logging before the function executes.
174     */
175    advice execution(acp::before_logging()) && !ignore() : before() {
176        std::string msg;
177        build_before_msg(tjp->signature(), msg);
178        _printer->print(msg);
179        _printer->print('\n');
180    }
181
182};
