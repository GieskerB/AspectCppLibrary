 7namespace acp {
 8     /**
 9     * \attribute acp::null_check
10     * \brief Triggers an nullptr check for all pointer dereferencing operators
11     *
12     * This attribute marks a join point, on which the Nullptrcheck aspect should be applied.
13     */
14    attribute null_check();
15
16    /**
17     * \attribute acp::ignore_null_check
18     * \brief This attribute can be used to exclude certain join points form being affected by this aspect.
19     *
20     * If a scope already is annotated with this aspect's attribute, individual join points within can be
21     * excluded with this attribute.
22     */
23    attribute ignore_null_check();
24}

26/**
27 * \aspect NullptrCheck
28 * \file /src/bug_detection/nullptr_check.ah
29 * \brief Aspect that checks for null pointer dereferences.
30 *
31 * This aspect is used to detect an nullptr dereferencing, which would lead to a segfault.
32 * In the case that such an potential segfault an exception will be triggered and the
33 * dereferencing will not be performed.
34 * Unfortunately, it does not work with the -> operator, due to limitations in the AspectC++
35 * language and the way the builtin 'operator->' is defined.
36 *
37 * Make sure to use the '--builtin_operators' console argument when using this aspect
38 *
39 * \see acp::NullptrException
40 *
41 */
42aspect [[acp::middle_layer]] NullptrCheck : public CoreAspect {
43
44    /**
45     * \pointcut dereferencing_pointer
46     * \brief The set of all builtin pointer dereferencing join points.
47     */
48    pointcut dereferencing_pointer() = builtin("%& operator *(%*)");
49
50public:
51
52    NullptrCheck() {}
53    virtual ~NullptrCheck() = default;
54
55    /**
56     * \pointcut where
57     * \see :ref:`CoreAspect::where <core_aspect_ah_where>`
58     */
59    pointcut virtual where() = within(acp::null_check());
60
61    /**
62     * \pointcut ignore
63     * \see :ref:`CoreAspect::ignore <core_aspect_ah_ignore>`
64     */
65    pointcut virtual ignore() = within("NullptrCheck") || within("acp") || within(acp::ignore_null_check()) || within(acp::ignore_everything());
66
67    /**
68     * \advice nullptr_check
69     * \brief This Advice checks if a pointer is null before dereferencing it.
70     */
71    advice where() && dereferencing_pointer() && !ignore(): before() {
72        // Just casting to int* to check if the pointer is null.
73        if (*((int *)tjp->arg(0)) == 0) {
74            throw acp::NullptrException(tjp->signature(), tjp->line());
75        }
76    }
77
78
79};
